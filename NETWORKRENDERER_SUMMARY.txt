USAGE EXAMPLE
================================================================================

# Setup
from network_renderer import NetworkRenderer, create_network_visualization
from publication_viz_core import Typography
import matplotlib.pyplot as plt

# Configure matplotlib
Typography.configure_mpl()

# Create figure
fig, ax = create_network_visualization(fig_width=12, fig_height=12)

# Create renderer
renderer = NetworkRenderer(ax, title="IoT Mesh Network")
renderer.initialize()

# Animation loop
for frame in range(num_frames):
    # Prepare simulation state
    state = {
        'nodes': nodes_dict,
        'edges': edges_dict,
        'infected_nodes': infected_set,
        'exposed_nodes': exposed_set,
        'active_route': route_list,
        'selected_qlearn_routes': qlearn_routes,
        'failed_routes': failed_set,
    }
    
    # Update and render
    artists = renderer.update(frame, state)
    plt.draw()

plt.show()
